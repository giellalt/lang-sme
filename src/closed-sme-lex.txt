! Interrogative pronouns
! ----------------------

!Giving ideosyncratic Sg Nom lexically
!Sending the oblique forms to an oblique sublexicon
LEXICON Interrogative
mii+Pron+Interr+Sg+Nom:mij K-son ;
gii+Pron+Interr+Sg+Nom:gij K-son ;
mii+Pron+Interr:mo oblintercas ;

!Then going diractly to the case paradigm of the demonstratives
LEXICON oblintercas 
 +Sg:0 obldemcassg ;
 +Pl:0 obldemcaspl ;

!Giving baseform + all demonstrative stems
!Pointing to case paradigms
LEXICON Demonstrative

!Splitting into Sg, Pl and Ess, different paradigms
LEXICON demcas
 +Sg:0 demcaspl ;
 +Pl:0 demcassg ;
 +Ess:nin K ;

!Splitting into oblique and nonoblique
!This is because of the interrogatives
LEXICON demcassg

LEXICON demcaspl
