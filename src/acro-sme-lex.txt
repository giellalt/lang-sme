%- ARABIC ;     !^C^       ! Here we dive over to the numerals, 
    ARABIC ;    !^C^       ! Here we dive over to the numerals
+N+ACR: acroconnector ;    ! Here we add the POS tag and go ACRO

LEXICON acroconnector      ! Here comes a set of possible symbols to
    acrosuff ;!SUB         ! put between the abbreviation and its suffix
:%: acrosuff1 ;
:%  acrosuff2 ;            ! The lexica are split in order to avoid the POS
:%' acrosuff ;!SUB         ! tag for other symbols using the acroconnector
:%` acrosuff ;!SUB         ! lexicon. (cf. the punct-sme-lex.txt file)
:%- RAbbr ;

LEXICON acrosuff
+Sg+Nom: K ;
+Sg+Gen: K ;

+Sg+Acc:a K ; !SUB
+Sg+Ill:ij K ; !SUB

+Sg+Loc:s K ;   ! NRK:s
