LEXICON acroconnector      ! Here comes a set of possible symbols to
    acrosuff ;!SUB         ! put between the abbreviation and its suffix
:%: acrosuff ;             ! The lexica are split in order to avoid the POS
:%' acrosuff ;!SUB         ! tag for other symbols using the acroconnector
:%` acrosuff ;!SUB         ! lexicon. (cf. the punct-sme-lex.txt file)
:%- R ; !^C^
